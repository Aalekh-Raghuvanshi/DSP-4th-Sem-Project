# ─────────────────────────────────────────────────────────────────────────────
# NeuralGate · Face Authentication Backend
# Python 3.10+ required
#
# Install (CPU only):
#   pip install -r requirements.txt
#
# Install (NVIDIA GPU — faster inference):
#   pip install -r requirements.txt --extra-index-url https://aiinfra.pkgs.visualstudio.com/PublicPackages/_packaging/onnxruntime-cuda-12/pypi/simple/
#   Then replace onnxruntime with onnxruntime-gpu==1.21.0 below.
# ─────────────────────────────────────────────────────────────────────────────

# ── Web framework ─────────────────────────────────────────────────────────────
fastapi
uvicorn[standard]      # ASGI server (includes uvloop, watchfiles)
python-multipart      # required for FastAPI Form / File uploads

# ── Face analysis ─────────────────────────────────────────────────────────────
insightface            # ArcFace + RetinaFace (latest stable)
onnxruntime        # CPU inference backend for InsightFace models
                               # → swap for onnxruntime-gpu==1.24.4 for CUDA

# ── Computer vision ───────────────────────────────────────────────────────────
opencv-python-headless   # OpenCV without GUI (server-safe)  
numpy                  # Array ops; InsightFace requires >=1.20

# ── Data validation ───────────────────────────────────────────────────────────
pydantic              # Used by FastAPI for request/response models
